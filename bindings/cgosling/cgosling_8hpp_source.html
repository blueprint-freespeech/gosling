40 template <typename T> class out_unique_ptr {

45  out_unique_ptr &operator=(out_unique_ptr &&) = delete;
46 
47  out_unique_ptr(out_unique_ptr &&that) : t_(that.t_), u_(that.u_) {
48  that.t_ = nullptr;
49  that.u_ = nullptr;
50  };
51  explicit constexpr out_unique_ptr(std::unique_ptr<T> &u)
52  : t_(nullptr), u_(&u) {}
53 
54  ~out_unique_ptr() {
55  if (u_ != nullptr) {
56  u_->reset(t_);
57  }
58  }
59 
60  constexpr operator T **() { return &t_; }
61 
62 private:
